    var m = /[^@]+@([\w\d\-\.]+)/.exec(email)
    return m && m[1]
  }

  function groupRecipients (recipients) {
    var groups = {}
    var host

      host = getHost(recipients[i]);
      (groups[host] || (groups[host] = [])).push(recipients[i])
    }
    return groups
  }
